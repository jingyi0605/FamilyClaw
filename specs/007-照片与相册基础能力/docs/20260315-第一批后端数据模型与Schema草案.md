from typing import Any, Literal

PhotoProviderStatus = Literal["draft", "active", "degraded", "disabled"]
PhotoSyncMode = Literal["full", "incremental", "asset", "album"]
PhotoSyncScope = Literal["all_assets", "album", "person"]

    include_albums: bool = True

    checksum: str | None = None
    album_refs: list[str] = Field(default_factory=list)
    people: list[ImmichConnectorPersonRecord] = Field(default_factory=list)
    metadata: dict[str, Any] = Field(default_factory=dict)


class ImmichConnectorSyncSummary(BaseModel):
    fetched: int = Field(ge=0)
    created: int = Field(ge=0)
    updated: int = Field(ge=0)
    skipped: int = Field(ge=0)
    failed: int = Field(ge=0)


class ImmichConnectorResponse(BaseModel):

    sync_summary: ImmichConnectorSyncSummary
    next_cursor: str | None = None
    warnings: list[dict[str, Any]] = Field(default_factory=list)
    errors: list[dict[str, Any]] = Field(default_factory=list)
